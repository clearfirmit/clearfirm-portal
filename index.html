        // Simple client-side router

        class Router {

            constructor() {

                this.routes = {};

                this.currentRoute = null;

                window.addEventListener('popstate', () => this.handleRoute());

                // Handle link clicks

                document.addEventListener('click', (e) => {

                    if (e.target.tagName === 'A' && e.target.getAttribute('href')?.startsWith('/')) {

                        e.preventDefault();

                        this.navigate(e.target.getAttribute('href'));

                    }

                });

            }


            addRoute(path, handler) {

                this.routes[path] = handler;

            }


            navigate(path) {

                window.history.pushState({}, '', path);

                this.handleRoute();

            }


            handleRoute() {

                const path = window.location.pathname;

                const handler = this.routes[path] || this.routes['/'];

                handler();

            }

        }

        function initializeAdminDashboard() {

            const uploadZone = document.getElementById('uploadZone');

            const fileInput = document.getElementById('fileInput');

            const uploadBtn = document.getElementById('uploadBtn');

            const manualForm = document.getElementById('manualEntryForm');

            uploadZone.addEventListener('click', () => fileInput.click());

            

            uploadZone.addEventListener('dragover', (e) => {

                e.preventDefault();

                uploadZone.classList.add('drag-over');

            });


            uploadZone.addEventListener('dragleave', () => {

                uploadZone.classList.remove('drag-over');

            });


            uploadZone.addEventListener('drop', (e) => {

                e.preventDefault();

                uploadZone.classList.remove('drag-over');

                handleFiles(e.dataTransfer.files);

            });


            fileInput.addEventListener('change', (e) => {

                handleFiles(e.target.files);

            });


            uploadBtn.addEventListener('click', uploadFiles);


            // Manual entry form

            manualForm.addEventListener('submit', handleManualEntry);

        }


        let selectedFiles = [];


        function handleFiles(files) {

            selectedFiles = Array.from(files).filter(file => {

                const isValid = file.type === 'application/pdf' || file.type === 'text/plain';

                const isSmallEnough = file.size <= 10 * 1024 * 1024; // 10MB

                return isValid && isSmallEnough;

            });


            updateFileList();

            document.getElementById('uploadBtn').disabled = selectedFiles.length === 0;

        }


        function updateFileList() {

            const fileList = document.getElementById('fileList');

            fileList.innerHTML = selectedFiles.map((file, index) => `

                <div class="flex items-center justify-between bg-gray-50 p-2 rounded">

                    <span class="text-sm">${file.name} (${(file.size / 1024).toFixed(1)} KB)</span>

                    <button onclick="removeFile(${index})" class="text-red-600 hover:text-red-800">

                        Remove

                    </button>

                </div>

            `).join('');

        }


        function removeFile(index) {

            selectedFiles.splice(index, 1);

            updateFileList();

            document.getElementById('uploadBtn').disabled = selectedFiles.length === 0;

        }


        async function uploadFiles() {

            if (selectedFiles.length === 0) return;


            const uploadBtn = document.getElementById('uploadBtn');

            uploadBtn.disabled = true;

            uploadBtn.textContent = 'Uploading...';


            for (const file of selectedFiles) {

                const formData = new FormData();

                formData.append('file', file);


                try {

                    await fetch(`${API_BASE_URL}/admin/upload`, {

                        method: 'POST',

                        headers: {

                            'Authorization': `Bearer ${localStorage.getItem('adminToken')}`

                        },

                        body: formData

                    });

                } catch (error) {

                    console.error('Upload error:', error);

                }

            }


            showAlert('Files uploaded successfully!', 'success');

            selectedFiles = [];

            updateFileList();

            uploadBtn.textContent = 'Upload Files';

            uploadBtn.disabled = true;

            loadAdminData();

        }

                showAlert('Connection error. Please try again.');

        }


        async function loadAdminData() {

            try {

                const response = await fetch(`${API_BASE_URL}/admin/dashboard`, {

                    headers: {

                        'Authorization': `Bearer ${localStorage.getItem('adminToken')}`

        function updateRequestsTable(requests) {

                tbody.innerHTML = `

                        </td>

                    </tr>

                `;

                return;

            }


            tbody.innerHTML = requests.map(request => `

                    <td class="px-6 py-4 whitespace-nowrap">${request.ein}</td>

                    <td class="px-6 py-4 whitespace-nowrap">

                    </td>

                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">

                    </td>

                                Process

                            </button>

                        ` : '-'}

                    </td>

                </tr>

            `).join('');

        }


        function getStatusClass(status) {

            switch (status) {

                case 'pending': return 'bg-yellow-100 text-yellow-800';

                case 'processing': return 'bg-blue-100 text-blue-800';

                case 'completed': return 'bg-green-100 text-green-800';

                case 'failed': return 'bg-red-100 text-red-800';

                default: return 'bg-gray-100 text-gray-800';

            }

        }


        async function processRequest(requestId) {

            if (!confirm('Process this verification request?')) return;


            try {

                const response = await fetch(`${API_BASE_URL}/admin/process/${requestId}`, {

                    method: 'POST',

                    headers: {

                        'Authorization': `Bearer ${localStorage.getItem('adminToken')}`

                    }

                });


                if (response.ok) {

                    showAlert('Processing started!', 'success');

                    loadAdminData();

                } else {

                    showAlert('Failed to process request');

                }

            } catch (error) {

                console.error('Process error:', error);

                showAlert('Connection error. Please try again.');

            }

        }


        // Utility function for downloading reports

        async function downloadReport(ein) {

            try {

                const response = await fetch(`${API_BASE_URL}/reports/${ein}`, {

                    headers: {

                        'Authorization': `Bearer ${localStorage.getItem('token')}`

                    }

                });


                if (response.ok) {

                    const blob = await response.blob();

                    const url = window.URL.createObjectURL(blob);

                    const a = document.createElement('a');

                    a.href = url;

                    a.download = `business-report-${ein}.pdf`;

                    document.body.appendChild(a);

                    a.click();

                    document.body.removeChild(a);

                    window.URL.revokeObjectURL(url);

                } else {

                    showAlert('Failed to download report');

                }

            } catch (error) {

                console.error('Download error:', error);

                showAlert('Failed to download report');

            }

        }
